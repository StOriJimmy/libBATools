set(SRCS_FILES ${HEADERS} ${SOURCES})

if (OPTION_USE_BATOOLS_AS_DLL)

else(OPTION_USE_BATOOLS_AS_DLL)
	add_library (${PROJECT_NAME} STATIC	${SRCS_FILES})
endif(OPTION_USE_BATOOLS_AS_DLL)

	${LIB_PATH})
